                interface PokemonListItemOwnProps {
                	readonly pokemon: Pokemon;

                const getTypeList = (type: string, index: number, list: string[]) =>
                	`${type}${list.length > 1 && index < list.length - 1 ? ", " : ""}`;

                	};

                			flexDirection="column"
